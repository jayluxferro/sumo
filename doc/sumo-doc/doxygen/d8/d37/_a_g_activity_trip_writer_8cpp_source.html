62 AGActivityTripWriter::addTrip(const AGTrip& trip) {
63  int time = (trip.getDay() - 1) * 86400 + trip.getTime();
64 
65  myTripOutput.openTag(SUMO_TAG_TRIP)
66  .writeAttr(SUMO_ATTR_ID, trip.getVehicleName())
67  .writeAttr(SUMO_ATTR_TYPE, trip.getType())
68  .writeAttr(SUMO_ATTR_DEPART, time)
69  .writeAttr(SUMO_ATTR_DEPARTPOS, trip.getDep().getPosition())
70  .writeAttr(SUMO_ATTR_ARRIVALPOS, trip.getArr().getPosition())
71  .writeAttr(SUMO_ATTR_ARRIVALSPEED, 0.)
72  .writeAttr(SUMO_ATTR_FROM, trip.getDep().getStreet().getID());
73 
74  if (!trip.getPassed()->empty()) {
75  std::ostringstream oss;
76  for (std::list<AGPosition>::const_iterator it = trip.getPassed()->begin(); it != trip.getPassed()->end(); ++it) {
77  if (it != trip.getPassed()->begin()) {
78  oss << " ";
79  }
80  oss << it->getStreet().getID();
81  }
82  myTripOutput.writeAttr(SUMO_ATTR_VIA, oss.str());
83  }
84  myTripOutput.writeAttr(SUMO_ATTR_TO, trip.getArr().getStreet().getID());
85  myTripOutput.closeTag();
86 }

55 SUMOPolygon::writeXML(OutputDevice& out, bool geo) {
56  out.openTag(SUMO_TAG_POLY);
57  out.writeAttr(SUMO_ATTR_ID, StringUtils::escapeXML(getID()));
58  if (getType().size() > 0) {
59  out.writeAttr(SUMO_ATTR_TYPE, StringUtils::escapeXML(getType()));
60  }
61  out.writeAttr(SUMO_ATTR_COLOR, getColor());
62  out.writeAttr(SUMO_ATTR_FILL, getFill());
63  out.writeAttr(SUMO_ATTR_LAYER, getLayer());
64  PositionVector shape = getShape();
65  if (geo) {
66  out.writeAttr(SUMO_ATTR_GEO, true);
67  for (int i = 0; i < (int) shape.size(); i++) {
68  GeoConvHelper::getFinal().cartesian2geo(shape[i]);
69  }
70  }
71  out.writeAttr(SUMO_ATTR_SHAPE, shape);
72  if (getNaviDegree() != Shape::DEFAULT_ANGLE) {
73  out.writeAttr(SUMO_ATTR_ANGLE, getNaviDegree());
74  }
75  if (getImgFile() != Shape::DEFAULT_IMG_FILE) {
76  out.writeAttr(SUMO_ATTR_IMGFILE, getImgFile());
77  }
78  writeParams(out);
79  out.closeTag();
80 }

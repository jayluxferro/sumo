77  // check whether an "all" indicator is there
78  bool hadAll = false;
79  for (std::vector<int>::const_iterator i = myLanes.begin(); !hadAll && i != myLanes.end(); ++i) {
80  if ((*i) == -1) {
81  hadAll = true;
82  }
83  }
84  // no -> return

86  return;
87  }
88  // patch lane indices
89  myLanes.clear();

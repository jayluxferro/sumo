47 void
48 MSBatteryExport::write(OutputDevice& of, SUMOTime timestep, int precision) {
49  of.openTag(SUMO_TAG_TIMESTEP).writeAttr(SUMO_ATTR_TIME, time2string(timestep));
50  of.setPrecision(precision);
51 
52  MSVehicleControl& vc = MSNet::getInstance()->getVehicleControl();
53  MSVehicleControl::constVehIt it = vc.loadedVehBegin();
54  MSVehicleControl::constVehIt end = vc.loadedVehEnd();
55  for (; it != end; ++it) {
56  const MSVehicle* veh = static_cast<const MSVehicle*>((*it).second);
57 
58  if (!veh->isOnRoad()) {
59  continue;
60  }
61 
62  std::string fclass = veh->getVehicleType().getID();
63  fclass = fclass.substr(0, fclass.find_first_of("@"));
64 
65  Position pos = veh->getLane()->getShape().positionAtOffset(veh->getPositionOnLane());
66 
67  if (static_cast<MSDevice_Battery*>(veh->getDevice(typeid(MSDevice_Battery))) != 0) {
68  MSDevice_Battery* batteryToExport = dynamic_cast<MSDevice_Battery*>(veh->getDevice(typeid(MSDevice_Battery)));
69  if (batteryToExport->getMaximumBatteryCapacity() > 0) {
70  // Open Row
71  of.openTag(SUMO_TAG_VEHICLE);
72  // Write ID
73  of.writeAttr(SUMO_ATTR_ID, veh->getID());
74  // Write consum
75  of.writeAttr(SUMO_ATTR_ENERGYCONSUMED, batteryToExport->getConsum());
76  // Write Actual battery capacity
77  of.writeAttr(SUMO_ATTR_ACTUALBATTERYCAPACITY, batteryToExport->getActualBatteryCapacity());
78  // Write Maximum battery capacity
79  of.writeAttr(SUMO_ATTR_MAXIMUMBATTERYCAPACITY, batteryToExport->getMaximumBatteryCapacity());
80  // Write Charging Station ID
81  of.writeAttr(SUMO_ATTR_CHARGINGSTATIONID, batteryToExport->getChargingStationID());
82  // Write Charge charged in the Battery
83  of.writeAttr(SUMO_ATTR_ENERGYCHARGED, batteryToExport->getEnergyCharged());
84  // Write ChargeInTransit
85  if (batteryToExport->isChargingInTransit()) {
86  of.writeAttr(SUMO_ATTR_ENERGYCHARGEDINTRANSIT, batteryToExport->getEnergyCharged());
87  } else {
88  of.writeAttr(SUMO_ATTR_ENERGYCHARGEDINTRANSIT, 0.00);
89  }
90  // Write ChargingStopped
91  if (batteryToExport->isChargingStopped()) {
92  of.writeAttr(SUMO_ATTR_ENERGYCHARGEDSTOPPED, batteryToExport->getEnergyCharged());
93  } else {
94  of.writeAttr(SUMO_ATTR_ENERGYCHARGEDSTOPPED, 0.00);
95  }
96  // Write Speed
97  of.writeAttr(SUMO_ATTR_SPEED, veh->getSpeed());
98  // Write Acceleration
99  of.writeAttr(SUMO_ATTR_ACCELERATION, veh->getAcceleration());
100  // Write pos x
101  of.writeAttr(SUMO_ATTR_X, veh->getPosition().x());
102  // Write pos y
103  of.writeAttr(SUMO_ATTR_Y, veh->getPosition().y());
104  // Write Lane ID
105  of.writeAttr(SUMO_ATTR_LANE, veh->getLane()->getID());
106  // Write vehicle position in the lane
107  of.writeAttr(SUMO_ATTR_POSONLANE, veh->getPositionOnLane());
108  // Write Time stopped (In all cases)
109  of.writeAttr(SUMO_ATTR_TIMESTOPPED, batteryToExport->getVehicleStopped());
110  // Close Row
111  of.closeTag();
112  }
113  }
114  }
115  of.closeTag();
116 }

312 MSNet::SimulationState
313 MSNet::simulate(SUMOTime start, SUMOTime stop) {
314  // report the begin when wished
315  WRITE_MESSAGE("Simulation started with time: " + time2string(start));
316  // the simulation loop
317  SimulationState state = SIMSTATE_RUNNING;
318  // state loading may have changed the start time so we need to reinit it
319  myStep = start;
320 #ifndef NO_TRACI
321 #ifdef HAVE_PYTHON
322  if (OptionsCont::getOptions().isSet("python-script")) {
323  TraCIServer::runEmbedded(OptionsCont::getOptions().getString("python-script"));
324  closeSimulation(start);
325  WRITE_MESSAGE("Simulation ended at time: " + time2string(getCurrentTimeStep()));
326  WRITE_MESSAGE("Reason: Script ended");
327  return state;
328  }
329 #endif
330 #endif
331  while (state == SIMSTATE_RUNNING) {
332  if (myLogStepNumber) {
333  preSimStepOutput();
334  }
335  simulationStep();
336  if (myLogStepNumber) {
337  postSimStepOutput();
338  }
339  state = simulationState(stop);
340 #ifdef DEBUG_SIMSTEP
341  std::cout << SIMTIME << " MSNet::simulate(" << start << ", " << stop << ")"
342  << "\n simulation state: " << getStateMessage(state)
343  << std::endl;
344 #endif
345 #ifndef NO_TRACI
346  if (state == SIMSTATE_LOADING) {
347  OptionsIO::setArgs(TraCIServer::getInstance()->getLoadArgs());
348  TraCIServer::getInstance()->getLoadArgs().clear();
349  } else if (state != SIMSTATE_RUNNING) {
350  if (TraCIServer::getInstance() != 0 && !TraCIServer::wasClosed()) {
351  // overrides SIMSTATE_END_STEP_REACHED, e.g. (TraCI ignore SUMO's --end option)
352  state = SIMSTATE_RUNNING;
353  }
354  }
355 #endif
356  }
357  // report the end when wished
358  WRITE_MESSAGE("Simulation ended at time: " + time2string(getCurrentTimeStep()));
359  WRITE_MESSAGE("Reason: " + getStateMessage(state));
360  // exit simulation loop
361  closeSimulation(start);
362  return state;
363 }

438 void
439 MSNet::simulationStep() {
440 #ifdef DEBUG_SIMSTEP
441  std::cout << SIMTIME << ": MSNet::simulationStep() called"
442  << ", myStep = " << myStep
443  << std::endl;
444 #endif
445 #ifndef NO_TRACI
446  if (myLogExecutionTime) {
447  myTraCIStepDuration = SysUtils::getCurrentMillis();
448  }
449  TraCIServer* t = TraCIServer::getInstance();
450  if (t != 0 && !t->isEmbedded()) {
451  t->processCommandsUntilSimStep(myStep);
452 #ifdef DEBUG_SIMSTEP
453  bool loadRequested = !TraCI::getLoadArgs().empty();
454  bool closed = TraCIServer::wasClosed();
455  assert(t->getTargetTime() >= myStep || loadRequested || closed);
456 #endif
457  }
458  if (myLogExecutionTime) {
459  myTraCIStepDuration = SysUtils::getCurrentMillis() - myTraCIStepDuration;
460  }
461 #ifdef DEBUG_SIMSTEP
462  std::cout << SIMTIME << ": TraCI target time: " << t->getTargetTime() << std::endl;
463 #endif
464 #endif
465  // execute beginOfTimestepEvents
466  if (myLogExecutionTime) {
467  mySimStepDuration = SysUtils::getCurrentMillis();
468  }
469  // simulation state output
470  std::vector<SUMOTime>::iterator timeIt = find(myStateDumpTimes.begin(), myStateDumpTimes.end(), myStep);
471  if (timeIt != myStateDumpTimes.end()) {
472  const int dist = (int)distance(myStateDumpTimes.begin(), timeIt);
473  MSStateHandler::saveState(myStateDumpFiles[dist], myStep);
474  }
475  if (myStateDumpPeriod > 0 && myStep % myStateDumpPeriod == 0) {
476  MSStateHandler::saveState(myStateDumpPrefix + "_" + time2string(myStep) + myStateDumpSuffix, myStep);
477  }
478  myBeginOfTimestepEvents->execute(myStep);
479 #ifdef HAVE_FOX
480  MSDevice_Routing::waitForAll();
481 #endif
482  if (MSGlobals::gCheck4Accidents) {
483  myEdges->detectCollisions(myStep, STAGE_EVENTS);
484  }
485  // check whether the tls programs need to be switched
486  myLogics->check2Switch(myStep);
487 
488  if (MSGlobals::gUseMesoSim) {
489  MSGlobals::gMesoNet->simulate(myStep);
490  } else {
491  // assure all lanes with vehicles are 'active'
492  myEdges->patchActiveLanes();
493 
494  // compute safe velocities for all vehicles for the next few lanes
495  // also register ApproachingVehicleInformation for all links
496  myEdges->planMovements(myStep);
497 
498  // decide right-of-way and execute movements
499  myEdges->executeMovements(myStep);
500  if (MSGlobals::gCheck4Accidents) {
501  myEdges->detectCollisions(myStep, STAGE_MOVEMENTS);
502  }
503 
504  // vehicles may change lanes
505  myEdges->changeLanes(myStep);
506 
507  if (MSGlobals::gCheck4Accidents) {
508  myEdges->detectCollisions(myStep, STAGE_LANECHANGE);
509  }
510  }
511  loadRoutes();
512 
513  // persons
514  if (myPersonControl != 0 && myPersonControl->hasTransportables()) {
515  myPersonControl->checkWaiting(this, myStep);
516  }
517  // containers
518  if (myContainerControl != 0 && myContainerControl->hasTransportables()) {
519  myContainerControl->checkWaiting(this, myStep);
520  }
521  // insert vehicles
522  myInserter->determineCandidates(myStep);
523  myInsertionEvents->execute(myStep);
524 #ifdef HAVE_FOX
525  MSDevice_Routing::waitForAll();
526 #endif
527  myInserter->emitVehicles(myStep);
528  if (MSGlobals::gCheck4Accidents) {
529  //myEdges->patchActiveLanes(); // @note required to detect collisions on lanes that were empty before insertion. wasteful?
530  myEdges->detectCollisions(myStep, STAGE_INSERTIONS);
531  }
532  MSVehicleTransfer::getInstance()->checkInsertions(myStep);
533 
534  // execute endOfTimestepEvents
535  myEndOfTimestepEvents->execute(myStep);
536 
537 #ifndef NO_TRACI
538  if (TraCIServer::getInstance() != 0) {
539  if (myLogExecutionTime) {
540  myTraCIStepDuration -= SysUtils::getCurrentMillis();
541  }
542  libsumo::Helper::postProcessRemoteControl();
543  if (myLogExecutionTime) {
544  myTraCIStepDuration += SysUtils::getCurrentMillis();
545  }
546  }
547 #endif
548  // update and write (if needed) detector values
549  writeOutput();
550 
551  if (myLogExecutionTime) {
552  mySimStepDuration = SysUtils::getCurrentMillis() - mySimStepDuration;
553  myVehiclesMoved += myVehicleControl->getRunningVehicleNo();
554  }
555  myStep += DELTA_T;
556 }

559 MSNet::SimulationState
560 MSNet::simulationState(SUMOTime stopTime) const {
561 #ifndef NO_TRACI
562  if (TraCIServer::wasClosed()) {
563  return SIMSTATE_CONNECTION_CLOSED;
564  }
565  if (TraCIServer::getInstance() != 0 && !TraCIServer::getInstance()->getLoadArgs().empty()) {
566  return SIMSTATE_LOADING;
567  }
568  if ((stopTime < 0 || myStep > stopTime) && TraCIServer::getInstance() == 0) {
569 #else
570  if (stopTime < 0 || myStep > stopTime) {
571 #endif
572  if (myInsertionEvents->isEmpty()
573  && (myVehicleControl->getActiveVehicleCount() == 0)
574  && (myInserter->getPendingFlowCount() == 0)
575  && (myPersonControl == 0 || !myPersonControl->hasNonWaiting())
576  && (myContainerControl == 0 || !myContainerControl->hasNonWaiting())) {
577  if (myPersonControl) {
578  myPersonControl->abortWaitingForVehicle();
579  }
580  if (myContainerControl) {
581  myContainerControl->abortWaitingForVehicle();
582  }
583  myVehicleControl->abortWaiting();
584  return SIMSTATE_NO_FURTHER_VEHICLES;
585  }
586  }
587  if (stopTime >= 0 && myStep >= stopTime) {
588  return SIMSTATE_END_STEP_REACHED;
589  }
590  if (myMaxTeleports >= 0 && myVehicleControl->getTeleportCount() > myMaxTeleports) {
591  return SIMSTATE_TOO_MANY_TELEPORTS;
592  }
593  return SIMSTATE_RUNNING;
594 }

598 MSNet::getStateMessage(MSNet::SimulationState state) {
599  switch (state) {
600  case MSNet::SIMSTATE_RUNNING:
601  return "";
602  case MSNet::SIMSTATE_END_STEP_REACHED:
603  return "The final simulation step has been reached.";
604  case MSNet::SIMSTATE_NO_FURTHER_VEHICLES:
605  return "All vehicles have left the simulation.";
606  case MSNet::SIMSTATE_CONNECTION_CLOSED:
607  return "TraCI requested termination.";
608  case MSNet::SIMSTATE_ERROR_IN_SIM:
609  return "An error occured (see log).";
610  case MSNet::SIMSTATE_TOO_MANY_TELEPORTS:
611  return "Too many teleports.";
612  case MSNet::SIMSTATE_LOADING:
613  return "TraCI issued load command.";
614  default:
615  return "Unknown reason.";
616  }

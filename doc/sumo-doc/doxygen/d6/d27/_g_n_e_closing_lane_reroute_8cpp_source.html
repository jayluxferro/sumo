68 void
69 GNEClosingLaneReroute::writeClosingLaneReroute(OutputDevice& device) const {
70  // open closing reroute tag
71  device.openTag(getTag());
72  // write Lane ID
73  device.writeAttr(SUMO_ATTR_ID, myClosedLane->getID());
74  // write Allowed vehicles
75  device.writeAttr(SUMO_ATTR_ALLOW, getVehicleClassNames(myAllowedVehicles));
76  // write disallowed vehicles
77  device.writeAttr(SUMO_ATTR_DISALLOW, getVehicleClassNames(myDisallowedVehicles));
78  // close closing reroute tag
79  device.closeTag();
80 }

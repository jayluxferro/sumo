47 SUMOSAXAttributesImpl_Binary::SUMOSAXAttributesImpl_Binary(
48  const std::map<int, std::string>& predefinedTagsMML,
49  const std::string& objectType,
50  BinaryInputDevice* in, const char version) : SUMOSAXAttributes(objectType), myAttrIds(predefinedTagsMML) {
51  while (in->peek() == BinaryFormatter::BF_XML_ATTRIBUTE) {
52  int attr;
53  unsigned char attrByte;
54  *in >> attrByte;
55  attr = attrByte;
56  if (version > 1) {
57  in->putback(BinaryFormatter::BF_BYTE);
58  *in >> attrByte;
59  attr += 256 * attrByte;
60  }
61  int type = in->peek();
62  switch (type) {
63  case BinaryFormatter::BF_BYTE:
64  *in >> myCharValues[attr];
65  break;
66  case BinaryFormatter::BF_INTEGER:
67  *in >> myIntValues[attr];
68  break;
69  case BinaryFormatter::BF_FLOAT:
70  case BinaryFormatter::BF_SCALED2INT:
71  *in >> myFloatValues[attr];
72  break;
73  case BinaryFormatter::BF_STRING:
74  *in >> myStringValues[attr];
75  break;
76  case BinaryFormatter::BF_LIST: {
77  int size;
78  *in >> size;
79  while (size > 0) {
80  const int type = in->peek();
81  if (type != BinaryFormatter::BF_POSITION_2D && type != BinaryFormatter::BF_POSITION_3D &&
82  type != BinaryFormatter::BF_SCALED2INT_POSITION_2D &&
83  type != BinaryFormatter::BF_SCALED2INT_POSITION_3D) {
84  throw ProcessError("Invalid binary file, only supporting position vectors.");
85  }
86  size--;
87  Position p;
88  *in >> p;
89  myPositionVectors[attr].push_back(p);
90  }
91  break;
92  }
93  case BinaryFormatter::BF_EDGE:
94  *in >> myIntValues[attr];
95  break;
96  case BinaryFormatter::BF_LANE:
97  *in >> myIntValues[attr];
98  in->putback(BinaryFormatter::BF_BYTE);
99  *in >> myCharValues[attr];
100  break;
101  case BinaryFormatter::BF_POSITION_2D:
102  case BinaryFormatter::BF_POSITION_3D:
103  case BinaryFormatter::BF_SCALED2INT_POSITION_2D:
104  case BinaryFormatter::BF_SCALED2INT_POSITION_3D: {
105  Position p;
106  *in >> p;
107  myPositionVectors[attr].push_back(p);
108  break;
109  }
110  case BinaryFormatter::BF_BOUNDARY: {
111  Position p;
112  *in >> p;
113  myPositionVectors[attr].push_back(p);
114  in->putback(BinaryFormatter::BF_POSITION_2D);
115  *in >> p;
116  myPositionVectors[attr].push_back(p);
117  break;
118  }
119  case BinaryFormatter::BF_COLOR:
120  *in >> myIntValues[attr];
121  break;
122  case BinaryFormatter::BF_NODE_TYPE:
123  *in >> myCharValues[attr];
124  break;
125  case BinaryFormatter::BF_EDGE_FUNCTION:
126  *in >> myCharValues[attr];
127  break;
128  case BinaryFormatter::BF_ROUTE: {
129  std::ostringstream into(std::ios::binary);
130  int size;
131  *in >> size;
132  FileHelpers::writeByte(into, BinaryFormatter::BF_ROUTE);
133  FileHelpers::writeInt(into, size);
134  if (size > 0) {
135  int intsToRead = size - 1;
136  int bitsOrEntry;
137  in->putback(BinaryFormatter::BF_INTEGER);
138  *in >> bitsOrEntry;
139  FileHelpers::writeInt(into, bitsOrEntry);
140  if (bitsOrEntry < 0) {
141  intsToRead = (-bitsOrEntry * (size - 1) - 1) / sizeof(int) / 8 + 2;
142  }
143  while (intsToRead > 0) {
144  in->putback(BinaryFormatter::BF_INTEGER);
145  *in >> bitsOrEntry;
146  FileHelpers::writeInt(into, bitsOrEntry);
147  intsToRead--;
148  }
149  }
150  myStringValues[attr] = into.str();
151  break;
152  }
153  default:
154  throw ProcessError("Binary file is invalid, attribute type is unknown.");
155  }
156  myAttrs.insert(attr);
157  }
158 }

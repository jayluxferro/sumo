61 void
62 NIXMLTypesHandler::myStartElement(int element,
63  const SUMOSAXAttributes& attrs) {
64  switch (element) {
65  case SUMO_TAG_TYPE: {
66  bool ok = true;
67  // get the id, report a warning if not given or empty...
68  myCurrentTypeID = attrs.get<std::string>(SUMO_ATTR_ID, 0, ok);
69  const char* const id = myCurrentTypeID.c_str();
70  const std::string defType = myTypeCont.knows(myCurrentTypeID) ? myCurrentTypeID : "";
71  const int priority = attrs.getOpt<int>(SUMO_ATTR_PRIORITY, id, ok, myTypeCont.getPriority(defType));
72  const int numLanes = attrs.getOpt<int>(SUMO_ATTR_NUMLANES, id, ok, myTypeCont.getNumLanes(defType));
73  const double speed = attrs.getOpt<double>(SUMO_ATTR_SPEED, id, ok, myTypeCont.getSpeed(defType));
74  const std::string allowS = attrs.getOpt<std::string>(SUMO_ATTR_ALLOW, id, ok, "");
75  const std::string disallowS = attrs.getOpt<std::string>(SUMO_ATTR_DISALLOW, id, ok, "");
76  const bool oneway = attrs.getOpt<bool>(SUMO_ATTR_ONEWAY, id, ok, myTypeCont.getIsOneWay(defType));
77  const bool discard = attrs.getOpt<bool>(SUMO_ATTR_DISCARD, id, ok, false);
78  const double width = attrs.getOpt<double>(SUMO_ATTR_WIDTH, id, ok, myTypeCont.getWidth(defType));
79  const double sidewalkWidth = attrs.getOpt<double>(SUMO_ATTR_SIDEWALKWIDTH, id, ok, myTypeCont.getSidewalkWidth(defType));
80  const double bikeLaneWidth = attrs.getOpt<double>(SUMO_ATTR_BIKELANEWIDTH, id, ok, myTypeCont.getBikeLaneWidth(defType));
81  if (!ok) {
82  return;
83  }
84  // build the type
85  SVCPermissions permissions = myTypeCont.getPermissions(defType);
86  if (allowS != "" || disallowS != "") {
87  permissions = parseVehicleClasses(allowS, disallowS);
88  }
89  myTypeCont.insert(myCurrentTypeID, numLanes, speed, priority, permissions, width, oneway, sidewalkWidth, bikeLaneWidth);
90  if (discard) {
91  myTypeCont.markAsToDiscard(myCurrentTypeID);
92  }
93  SumoXMLAttr myAttrs[] = {SUMO_ATTR_PRIORITY, SUMO_ATTR_NUMLANES, SUMO_ATTR_SPEED,
94  SUMO_ATTR_ALLOW, SUMO_ATTR_DISALLOW, SUMO_ATTR_ONEWAY,
95  SUMO_ATTR_DISCARD, SUMO_ATTR_WIDTH, SUMO_ATTR_SIDEWALKWIDTH, SUMO_ATTR_BIKELANEWIDTH
96  };
97  for (int i = 0; i < 10; i++) {
98  if (attrs.hasAttribute(myAttrs[i])) {
99  myTypeCont.markAsSet(myCurrentTypeID, myAttrs[i]);
100  }
101  }
102  break;
103  }
104  case SUMO_TAG_RESTRICTION: {
105  bool ok = true;
106  const SUMOVehicleClass svc = getVehicleClassID(attrs.get<std::string>(SUMO_ATTR_VCLASS, myCurrentTypeID.c_str(), ok));
107  const double speed = attrs.get<double>(SUMO_ATTR_SPEED, myCurrentTypeID.c_str(), ok);
108  if (ok) {
109  myTypeCont.addRestriction(myCurrentTypeID, svc, speed);
110  }
111  break;
112  }
113  default:
114  break;
115  }
116 }

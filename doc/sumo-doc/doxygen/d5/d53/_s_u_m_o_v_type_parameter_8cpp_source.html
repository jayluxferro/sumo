212 void
213 SUMOVTypeParameter::write(OutputDevice& dev) const {
214  if (onlyReferenced) {
215  return;
216  }
217  dev.openTag(SUMO_TAG_VTYPE);
218  dev.writeAttr(SUMO_ATTR_ID, id);
219  if (wasSet(VTYPEPARS_LENGTH_SET)) {
220  dev.writeAttr(SUMO_ATTR_LENGTH, length);
221  }
222  if (wasSet(VTYPEPARS_MINGAP_SET)) {
223  dev.writeAttr(SUMO_ATTR_MINGAP, minGap);
224  }
225  if (wasSet(VTYPEPARS_MAXSPEED_SET)) {
226  dev.writeAttr(SUMO_ATTR_MAXSPEED, maxSpeed);
227  }
228  if (wasSet(VTYPEPARS_PROBABILITY_SET)) {
229  dev.writeAttr(SUMO_ATTR_PROB, defaultProbability);
230  }
231  if (wasSet(VTYPEPARS_SPEEDFACTOR_SET)) {
232  dev.writeAttr(SUMO_ATTR_SPEEDFACTOR, speedFactor);
233  }
234  if (wasSet(VTYPEPARS_ACTIONSTEPLENGTH_SET)) {
235  // Note: action step length is only exposed in seconds to the user
236  dev.writeAttr(SUMO_ATTR_ACTIONSTEPLENGTH, STEPS2TIME(actionStepLength));
237  }
238  if (wasSet(VTYPEPARS_VEHICLECLASS_SET)) {
239  dev.writeAttr(SUMO_ATTR_VCLASS, toString(vehicleClass));
240  }
241  if (wasSet(VTYPEPARS_EMISSIONCLASS_SET)) {
242  dev.writeAttr(SUMO_ATTR_EMISSIONCLASS, PollutantsInterface::getName(emissionClass));
243  }
244  if (wasSet(VTYPEPARS_IMPATIENCE_SET)) {
245  if (impatience == -std::numeric_limits<double>::max()) {
246  dev.writeAttr(SUMO_ATTR_IMPATIENCE, "off");
247  } else {
248  dev.writeAttr(SUMO_ATTR_IMPATIENCE, impatience);
249  }
250  }
251  if (wasSet(VTYPEPARS_SHAPE_SET)) {
252  dev.writeAttr(SUMO_ATTR_GUISHAPE, getVehicleShapeName(shape));
253  }
254  if (wasSet(VTYPEPARS_WIDTH_SET)) {
255  dev.writeAttr(SUMO_ATTR_WIDTH, width);
256  }
257  if (wasSet(VTYPEPARS_HEIGHT_SET)) {
258  dev.writeAttr(SUMO_ATTR_HEIGHT, height);
259  }
260  if (wasSet(VTYPEPARS_COLOR_SET)) {
261  dev.writeAttr(SUMO_ATTR_COLOR, color);
262  }
263  if (wasSet(VTYPEPARS_OSGFILE_SET)) {
264  dev.writeAttr(SUMO_ATTR_OSGFILE, osgFile);
265  }
266  if (wasSet(VTYPEPARS_IMGFILE_SET)) {
267  dev.writeAttr(SUMO_ATTR_IMGFILE, imgFile);
268  }
269  if (wasSet(VTYPEPARS_PERSON_CAPACITY)) {
270  dev.writeAttr(SUMO_ATTR_PERSON_CAPACITY, personCapacity);
271  }
272  if (wasSet(VTYPEPARS_CONTAINER_CAPACITY)) {
273  dev.writeAttr(SUMO_ATTR_CONTAINER_CAPACITY, containerCapacity);
274  }
275  if (wasSet(VTYPEPARS_BOARDING_DURATION)) {
276  dev.writeAttr(SUMO_ATTR_BOARDING_DURATION, boardingDuration);
277  }
278  if (wasSet(VTYPEPARS_LOADING_DURATION)) {
279  dev.writeAttr(SUMO_ATTR_LOADING_DURATION, loadingDuration);
280  }
281  if (wasSet(VTYPEPARS_MAXSPEED_LAT_SET)) {
282  dev.writeAttr(SUMO_ATTR_MAXSPEED_LAT, maxSpeedLat);
283  }
284  if (wasSet(VTYPEPARS_LATALIGNMENT_SET)) {
285  dev.writeAttr(SUMO_ATTR_LATALIGNMENT, latAlignment);
286  }
287  if (wasSet(VTYPEPARS_MINGAP_LAT_SET)) {
288  dev.writeAttr(SUMO_ATTR_MINGAP_LAT, minGapLat);
289  }
290  if (wasSet(VTYPEPARS_LANE_CHANGE_MODEL_SET)) {
291  dev.writeAttr(SUMO_ATTR_LANE_CHANGE_MODEL, lcModel);
292  }
293  for (SubParams::const_iterator i = lcParameter.begin(); i != lcParameter.end(); ++i) {
294  dev.writeAttr(i->first, i->second);
295  }
296  for (SubParams::const_iterator i = jmParameter.begin(); i != jmParameter.end(); ++i) {
297  dev.writeAttr(i->first, i->second);
298  }
299  if (wasSet(VTYPEPARS_CAR_FOLLOW_MODEL)) {
300  dev.writeAttr(SUMO_ATTR_CAR_FOLLOW_MODEL, SUMOXMLDefinitions::CarFollowModels.getString(cfModel));
301  }
302  for (SubParams::const_iterator i = cfParameter.begin(); i != cfParameter.end(); ++i) {
303  dev.writeAttr(i->first, i->second);
304  }
305  writeParams(dev);
306  dev.closeTag();
307 }

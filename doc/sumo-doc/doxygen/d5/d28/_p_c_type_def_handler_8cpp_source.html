56 PCTypeDefHandler::myStartElement(int element,
57  const SUMOSAXAttributes& attrs) {
58  if (element == SUMO_TAG_POLYTYPE) {
59  bool ok = true;
60  // get the id, report an error if not given or empty...
61  std::string id = attrs.get<std::string>(SUMO_ATTR_ID, 0, ok);
62  if (!ok) {
63  return;
64  }
65  const double layer = attrs.getOpt<double>(SUMO_ATTR_LAYER, id.c_str(), ok, myOptions.getFloat("layer"));
66  const bool discard = attrs.getOpt<bool>(SUMO_ATTR_DISCARD, id.c_str(), ok, false);
67  const bool allowFill = attrs.getOpt<bool>(SUMO_ATTR_FILL, id.c_str(), ok, myOptions.getBool("fill"));
68  const std::string type = attrs.getOpt<std::string>(SUMO_ATTR_NAME, id.c_str(), ok, myOptions.getString("type"));
69  const std::string prefix = attrs.getOpt<std::string>(SUMO_ATTR_PREFIX, id.c_str(), ok, myOptions.getString("prefix"));
70  const std::string color = attrs.getOpt<std::string>(SUMO_ATTR_COLOR, id.c_str(), ok, myOptions.getString("color"));
71  // !!! what about error handling?
72  if (!myContainer.add(id, type, color, prefix, layer, discard, allowFill)) {
73  WRITE_ERROR("Could not add polygon type '" + id + "' (probably the id is already used).");
74  }
75  }
